setPattern("^(?!00)(?:[0-9]{1,3}|1000)$");    //0~1000 的  （名将、MVP）
setPattern("^500|(?!00|[5-9][0-9]{2})(?:[0-9]{1,3})$"); //0~500 的 （网速）
setPattern("^(?!00)(?:[0-9]{1,2})$"); //0~99 的 （掉线率）

QComboBox cb;
   cb.setEditable(true);

   QRegExpValidator rep;
   QRegExp ex;
   ex.setPattern("^(?!00)(?:[0-9]{1,2})$"); //0~99
   rep.setRegExp(ex);
   cb.setValidator(&rep);
   cb.show();
